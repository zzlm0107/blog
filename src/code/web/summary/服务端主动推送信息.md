---
title: 服务端主动推送信息
---

前后端通信有时需要服务端主动推送信息给前端，除了 **轮询** 外，目前还有两种主流实现方式：

## Server-Sent Events

Server-Sent Events (SSE) 这是 HTML5 提供的一种事件推送机制，客户端通过EventSource API发起请求后，服务器会保持长连接，并持续、单向地向客户端推送数据流。适用场景如实时通知，新闻推送，大模型实时输出等。

- 优点：实现简单，基于HTTP协议；浏览器原生支持EventSource API；内置自动断线重连机制；相比WebSocket更轻量，在单向推送场景下资源消耗更低。
- 缺点：仅支持服务器到客户端的单向通信；不支持二进制数据。

在技术实现上，需要前后端配合：

1. 后端方面，需要指定 Content-Type 为 `text/event-stream `,并且注意SSE数据格式为 `data: xxxx\n\n`，如果表示发送完毕还需要补充 `data: [DONE]\n\n`，如果我们需要传递其他的事件类型和数据，只需要服务端按格式写入 `event: 事件名\n` 表示事件名，紧接着写入对应的data数据。
2. 客户端方面，需要创建一个 EventSource 实例，指定服务器端的URL，即可开始监听服务器端推送的事件。服务端默认发送的SSE数据会触发 EventSource 的 message 事件；如果服务端发送了自定义事件，客户端也需要监听同名事件，才能收到对应的事件数据。

下面是一个大模型输出并处理的后端简单示例，使用express 框架：

```js :collapsed-lines=20
/* SSE格式返回，顺便通过自定义事件 "session-id" 实现了会话记录的缓存，从而完成大模型上下文记忆功能 */
const chatHistory = new Map(); // 消息记忆map
router.get('/AIChat', async (req, res) => {
    const msg = decodeURIComponent(req.query.msg);
    const SessionId = req.query.SessionId;
    const currentSessionId = SessionId || Date.now().toString();

    try {
        // 设置流式响应头
        res.setHeader('Content-Type', 'text/event-stream; charset=utf-8'); // SSE
        res.setHeader('Cache-Control', 'no-cache');
        res.setHeader('Connection', 'keep-alive');

        // 自定义事件：指定事件类型
        res.write(`event: session-id\n`); // 指定事件名为 "session-id"
        res.write(`data: ${currentSessionId}\n\n`); // 数据内容

        // 获取或初始化会话历史
        let history = chatHistory.get(currentSessionId) || [{ role: 'system', content: '你的名字叫阿伟' }];
        // 添加用户新消息到历史
        history.push({ role: 'user', content: msg });

        const completion = await openai.chat.completions.create({
            messages: history,
            model: 'deepseek-chat',
            stream: true,
        });

        let fullResponse = '';
        // 逐个接收并发送流数据
        for await (const chunk of completion) {
            const content = chunk.choices[0]?.delta?.content || '';
            if (content) {
                const SSEData = `data: ${content}\n\n`;
                res.write(SSEData);
                // res.write(content);
                fullResponse += content;
            }
        }
        // 将AI回复添加到历史记录
        history.push({ role: 'assistant', content: fullResponse });
        chatHistory.set(currentSessionId, history); // 更新历史记录

        res.write(`data: [DONE]\n\n`);
        res.end();
    } catch (error) {
        console.log(error);
    }
});
```

前端方面相关代码如下：

```js
const content = document.querySelector('.content');
const input = document.querySelector('input');

let SessionId = ''; // 会话记忆 id
let currentEventSource = null; // 保存当前的 EventSource 实例以便关闭

function sendMsg() {
    // 如果已有连接，先关闭（防止重复发送或内存泄漏）
    if (currentEventSource) currentEventSource.close();

    const userMsg = input.value;

    // 构建 URL：注意 EventSource 仅支持 GET，参数需拼接到 URL 中
    // 如果 SessionId 存在则带上，否则由后端生成通过第一次会话的自定义事件带过来
    let url = `http://localhost:8866/AIChat?msg=${encodeURIComponent(userMsg)}`;
    if (SessionId) url += `&SessionId=${SessionId}`;

    // 创建 EventSource 连接
    currentEventSource = new EventSource(url);
    // 监听自定义事件
    currentEventSource.addEventListener('session-id', function (event) {
        // 只接收 event: session-id 的消息
        console.log('Session ID message:', event.data);
        SessionId = event.data;
    });

    let resText = ''; // 大模型响应的全部内容
    // 监听默认消息事件
    currentEventSource.onmessage = function (event) {
        if (event.data === '[DONE]') {
            currentEventSource.close();
            currentEventSource = null;
            return;
        }

        // 累加内容并渲染 Markdown
        console.log(event.data);
        resText += event.data;
        content.innerHTML = marked.parse(resText);
    };
}
```

除了使用浏览器的 EventSource API，实际上我们可以直接通过 fetch + ReadableStream 处理；因为 EventSource 只支持 get请求，无法携带复杂的请求头或 body，且无法处理二进制数据，所以必要时我们需要手动解析流数据，下面是前端相关的具体示例代码：

```js
async function sendMsg() {
    const resp = await fetch('http://localhost:8866/AIChat' + `?SessionId=${SessionId}`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            msg: input.value,
        }),
    });

    let resText = '';
    SessionId = resp.headers.get('Session-Id'); // 从响应头获取会话id，需后端配合

    const reader = resp.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
        const { done, value } = await reader.read(); // 读取相应的流片段
        if (done) break;

        const text = decoder.decode(value); // 用文本解析对象解析流，得到编码后的文字
        resText += text;

        // 解析并更新内容
        content.innerHTML = marked.parse(resText);
    }
}
```
